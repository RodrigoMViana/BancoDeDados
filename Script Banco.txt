


create table aluno (
id_aluno serial primary key,
	nome varchar(40),
	telefone char(11),
	email varchar(40)
);

create table professor (
id_professor serial primary key,
	nome varchar(40),
	cpf char(11),
	telefone char(11),
	email varchar(40)
);

create table materia (
	materia varchar(40),
	aluno char(11),
	professor varchar(40),
	media char (2)
);
